# ============================================================
# Geovanni Flores Portfolio - Website Analytics Dashboard
# Data: Last 7 days | All data is aggregated & anonymous
# ============================================================

library(ggplot2)
library(scales)
library(gridExtra)
library(grid)

# ── Colour palette ──────────────────────────────────────────
blue      <- "#378ADD"
blue_lt   <- "#B5D4F4"
teal      <- "#1D9E75"
purple    <- "#7F77DD"
amber     <- "#EF9F27"
coral     <- "#D85A30"
bg        <- "#FFFFFF"
bg2       <- "#F6F6F4"
text_dark <- "#1A1A1A"
text_mid  <- "#6B6B66"
border    <- "#E0DED8"

base_theme <- theme_minimal(base_family = "sans") +
  theme(
    plot.background    = element_rect(fill = bg,  color = NA),
    panel.background   = element_rect(fill = bg2, color = NA),
    panel.grid.major   = element_line(color = border, linewidth = 0.4),
    panel.grid.minor   = element_blank(),
    axis.text          = element_text(color = text_mid, size = 9),
    axis.title         = element_blank(),
    plot.title         = element_text(color = text_dark, size = 11,
                                      face = "bold", margin = margin(b = 4)),
    plot.subtitle      = element_text(color = text_mid,  size = 9,
                                      margin = margin(b = 8)),
    plot.margin        = margin(12, 14, 12, 14),
    legend.position    = "bottom",
    legend.text        = element_text(color = text_mid, size = 8),
    legend.key.size    = unit(0.5, "lines")
  )

# ── 1. DAILY VISITORS line chart ────────────────────────────
days <- c("Apr 29","Apr 30","May 1","May 2","May 3","May 4","May 5")

visits <- data.frame(
  day   = factor(days, levels = days),
  week  = c(4, 5, 6, 7, 8, 10, 12),
  prev  = c(2, 2, 3, 2, 3,  3,  4)
)

p_line <- ggplot(visits, aes(x = day)) +
  geom_ribbon(aes(ymin = 0, ymax = week, group = 1),
              fill = blue, alpha = 0.10) +
  geom_line(aes(y = prev,  group = 1, colour = "Previous period",
                linetype = "Previous period"), linewidth = 1) +
  geom_line(aes(y = week,  group = 1, colour = "This week",
                linetype = "This week"), linewidth = 1.5) +
  geom_point(aes(y = week), colour = blue, size = 2.5, shape = 16) +
  scale_colour_manual(
    name   = NULL,
    values = c("This week" = blue, "Previous period" = blue_lt)
  ) +
  scale_linetype_manual(
    name   = NULL,
    values = c("This week" = "solid", "Previous period" = "dashed")
  ) +
  scale_y_continuous(breaks = seq(0, 14, 2), limits = c(0, 14)) +
  labs(title    = "Daily visitors — last 7 days",
       subtitle = "Upward trend with 188.9% growth vs prior period") +
  base_theme +
  theme(axis.text.x = element_text(angle = 30, hjust = 1),
        legend.position = "bottom",
        legend.direction = "horizontal")

# ── 2. TOP PAGES bar chart ───────────────────────────────────
pages <- data.frame(
  page  = factor(
    c("Home","Marketing Projects","Presentations","Dashboard","Analytics"),
    levels = rev(c("Home","Marketing Projects","Presentations","Dashboard","Analytics"))
  ),
  views = c(35, 8, 5, 4, 4)
)

p_pages <- ggplot(pages, aes(x = views, y = page)) +
  geom_col(fill = blue, width = 0.6) +
  geom_text(aes(label = views), hjust = -0.3,
            colour = text_dark, size = 3, fontface = "bold") +
  scale_x_continuous(limits = c(0, 42), expand = c(0, 0)) +
  labs(title    = "Top pages",
       subtitle = "Views in last 7 days") +
  base_theme +
  theme(panel.grid.major.y = element_blank(),
        panel.grid.major.x = element_line(color = border, linewidth = 0.3))

# ── 3. VISITORS BY COUNTRY bar chart ────────────────────────
countries <- data.frame(
  country = factor(
    c("United States","South Korea","France","India","Italy","Taiwan"),
    levels = rev(c("United States","South Korea","France","India","Italy","Taiwan"))
  ),
  users = c(20, 2, 1, 1, 1, 1)
)

p_countries <- ggplot(countries, aes(x = users, y = country)) +
  geom_col(fill = teal, width = 0.6) +
  geom_text(aes(label = users), hjust = -0.3,
            colour = text_dark, size = 3, fontface = "bold") +
  scale_x_continuous(limits = c(0, 24), expand = c(0, 0)) +
  labs(title    = "Visitors by country",
       subtitle = "Aggregated — no personal data") +
  base_theme +
  theme(panel.grid.major.y = element_blank(),
        panel.grid.major.x = element_line(color = border, linewidth = 0.3))

# ── 4. ENGAGEMENT EVENTS bar chart ──────────────────────────
events <- data.frame(
  event = factor(
    c("Page view","Scroll","Engagement","Click","Session start"),
    levels = c("Page view","Scroll","Engagement","Click","Session start")
  ),
  count  = c(30, 22, 16, 4, 1),
  colour = c(blue, teal, purple, amber, coral)
)

p_events <- ggplot(events, aes(x = event, y = count, fill = event)) +
  geom_col(width = 0.6, show.legend = FALSE) +
  geom_text(aes(label = count), vjust = -0.4,
            colour = text_dark, size = 3, fontface = "bold") +
  scale_fill_manual(values = setNames(events$colour, events$event)) +
  scale_y_continuous(limits = c(0, 36), expand = c(0, 0)) +
  labs(title    = "Engagement events — realtime snapshot",
       subtitle = "Event types captured anonymously via Google Analytics 4") +
  base_theme +
  theme(panel.grid.major.x = element_blank())

# ── 5. METRIC SUMMARY table ──────────────────────────────────
metrics_df <- data.frame(
  Metric = c("Active users", "New users", "Page views", "Avg. session"),
  Value  = c("~40",          "38",        "176",        "1m 42s"),
  Change = c("+188.9%",      "+177.8%",   "+158.8%",   "Growing")
)

p_metrics <- tableGrob(
  metrics_df,
  rows  = NULL,
  theme = ttheme_minimal(
    core    = list(fg_params = list(col = text_dark, fontsize = 9,
                                   hjust = 0, x = 0.05)),
    colhead = list(fg_params = list(col = text_mid,  fontsize = 9,
                                   fontface = "bold", hjust = 0, x = 0.05))
  )
)

# ── Assemble & save PNG ──────────────────────────────────────
png("analytics_dashboard.png", width = 1200, height = 1000, res = 120, bg = bg)

grid.newpage()
pushViewport(viewport(layout = grid.layout(
  nrow    = 4,
  ncol    = 2,
  widths  = unit(c(1, 1), "null"),
  heights = unit(c(0.15, 0.38, 0.28, 0.28), "null")
)))

# Header
pushViewport(viewport(layout.pos.row = 1, layout.pos.col = 1:2))
grid.rect(gp = gpar(fill = bg, col = NA))
grid.text("Website Analytics — Geovanni Flores Portfolio",
          x = 0.03, hjust = 0,
          gp = gpar(fontsize = 18, fontface = "bold", col = text_dark))
grid.text("Last 7 days  |  All data aggregated & anonymous  |  No personal information collected",
          x = 0.03, y = 0.3, hjust = 0,
          gp = gpar(fontsize = 9, col = text_mid))
popViewport()

# Line chart (full width)
print(p_line, vp = viewport(layout.pos.row = 2, layout.pos.col = 1:2))

# Pages + Countries
print(p_pages,     vp = viewport(layout.pos.row = 3, layout.pos.col = 1))
print(p_countries, vp = viewport(layout.pos.row = 3, layout.pos.col = 2))

# Events + Metrics table
print(p_events, vp = viewport(layout.pos.row = 4, layout.pos.col = 1))

pushViewport(viewport(layout.pos.row = 4, layout.pos.col = 2))
grid.rect(gp = gpar(fill = bg2, col = NA))
grid.text("Summary metrics",
          x = 0.05, y = 0.92, hjust = 0,
          gp = gpar(fontsize = 11, fontface = "bold", col = text_dark))
pushViewport(viewport(x = 0.5, y = 0.45, width = 0.9, height = 0.75))
grid.draw(p_metrics)
popViewport()
grid.text(
  "Privacy notice: All analytics collected in aggregate via Google Analytics 4. No PII tracked or stored.",
  x = 0.05, y = 0.06, hjust = 0,
  gp = gpar(fontsize = 7, col = text_mid)
)
popViewport()
